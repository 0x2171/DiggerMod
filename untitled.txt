-- мониторинг памяти
local lastMemoryCheckTime = 0
local MEMORY_CHECK_INTERVAL = 2  -- интервал проверки в секундах

-- Функция для вывода памяти в чат
function CheckAndReportMemory()
    local currentTime = CS.UnityEngine.Time.time

    if currentTime - lastMemoryCheckTime >= MEMORY_CHECK_INTERVAL then
        local memoryUsageKB = collectgarbage("count")  -- в килобайтах
        local memoryUsageMB = memoryUsageKB / 1024  -- переводим в мегабайты
        local formattedMemory = string.format("Память: %.1f МБ", memoryUsageMB)

        -- Определяем цвет в зависимости от потребления памяти
        local colorCode
        if memoryUsageMB < 100 then
            colorCode = "[a3fc95]"  -- зелёный: меньше 100 МБ
        elseif memoryUsageMB <= 300 then
            colorCode = "[fcd395]"  -- жёлтый: от 100 до 300 МБ (включительно)
        else
            colorCode = "[f54949]"  -- красный: больше 300 МБ
        end

        LogChatMessage(formattedMemory, colorCode)

        lastMemoryCheckTime = currentTime
    end
end

function createGradientTexture(width, height, startColor, endColor)
    local texture = UnityEngine.Texture2D(width, height)
    local pixels = {}
    local maxDistance = math.sqrt(width * width + height * height)  -- максимальная длина диагонали

    for y = 0, height - 1 do
        for x = 0, width - 1 do
            -- Расстояние от нижнего левого угла (0, height-1) до текущей точки (x, y)
            local distance = math.sqrt(x * x + (height - 1 - y) * (height - 1 - y))
            -- Нормализованный коэффициент (0..1)
            local t = distance / maxDistance

            -- Линейная интерполяция между цветами
            local color = UnityEngine.Color.Lerp(startColor, endColor, t)
            pixels[#pixels + 1] = color
        end
    end

    texture:SetPixels(pixels)
    texture:Apply()
    return texture
end

-- Вспомогательная функция для создания цветной текстуры с суффиксом 2
function createColorTexture2(color)
    local tex2 = UnityEngine.Texture2D(1, 1)
    tex2:SetPixel(0, 0, color)
    tex2:Apply()
    return tex2
end

local function RotateFormRight(form)
    return rotateRightMap[form] or form
end

    return rotateLeftMap[form] or form

local function RotateOffsetRight(offset)
    return CS.UnityEngine.Vector3(offset.z, offset.y, -offset.x)
end

local function RotateOffsetLeft(offset)
    return CS.UnityEngine.Vector3(-offset.z, offset.y, offset.x)
end

function RotateCopiedBlocksRight()
    if staticSelectionMode then
        Chat:AddMessage("В статичном режиме поворот отключён.", "[f54949]")
        return
    end
    buildData.source = "clipboard"
    if not buildData.blocks or #buildData.blocks == 0 then
        Chat:AddMessage("Сначала скопируй область кнопкой CopyArea", "[f54949]")
        return
    end

    -- Переиспользуем существующую таблицу вместо создания новой
    for i, data in ipairs(buildData.blocks) do
        buildData.blocks[i].offset = RotateOffsetRight(data.offset)
        buildData.blocks[i].blockForm = RotateFormRight(data.blockForm)
    end
    currentRotation = (currentRotation + 90) % 360
    Chat:AddMessage("Выделенная область повернута вправо на 90°", "[a455f2]")
end

local function RotatePointAroundY(pos, center, angleDegrees)
    local rad = math.rad(angleDegrees)
    local cosA = math.cos(rad)
    local sinA = math.sin(rad)
    local x = pos.x - center.x
    local z = pos.z - center.z
    local rotatedX = x * cosA - z * sinA
    local rotatedZ = x * sinA + z * cosA
    return CS.UnityEngine.Vector3(rotatedX + center.x, pos.y, rotatedZ + center.z)
end

function Teleport()
    if not Player.Camera then
        return false
    end

    local camera = Player.Camera
    local centerVec = CS.UnityEngine.Vector3(0.5, 0.5, 0)
    local ray = camera:ViewportPointToRay(centerVec)

    local hasHit, hit = Physics:Ray(ray, 999)

    if hasHit and hit.point then
        local blockX = math.floor(hit.point.x)
        local blockY = math.floor(hit.point.y)
        local blockZ = math.floor(hit.point.z)

        -- Получаем направление луча (взгляда игрока)
        local direction = ray.direction
        direction:Normalize()  -- нормализуем вектор

        -- Сдвигаем позицию на 1 блок *перед* блоком, в который попали
        local teleportPos = CS.UnityEngine.Vector3(
            blockX + 0.5 + direction.x,
            blockY + direction.y,
            blockZ + 0.5 + direction.z
        )

        Player.Position = teleportPos
        return true
    else
        return false
    end
end

local function GetAimPoint()
    -- Проверка камеры
    if not playerCamera or not playerCamera.transform then
        return nil
    end

    local centerVec = CS.UnityEngine.Vector3(0.5, 0.5, 0)
    local ray = playerCamera:ViewportPointToRay(centerVec)
    local terrainLayer = CS.UnityEngine.LayerMask.NameToLayer("Terrain")

    local hasHit, hit = Physics:Ray(ray, 999, terrainLayer) 

    if hasHit and hit.point and hit.normal then
        local normalOffset = hit.normal * 0.1
        local vecpoint = hit.point - normalOffset

        -- Округление координат
        return CS.UnityEngine.Vector3(
            math.floor(vecpoint.x),
            math.floor(vecpoint.y),
            math.floor(vecpoint.z)
        )
    end
end

local function AreAllSlidersAboveFive()
    local count = 0
    if blocksCountX > 5 then count = count + 1 end
    if blocksCountY > 5 then count = count + 1 end
    if blocksCountZ > 5 then count = count + 1 end
    return count >= 2
end

function SetBlock()
    if AreAllSlidersAboveFive() then
        return
    end

    if not Player.IsAdmin and not Player.IsModerator then
        return  -- Выходим, если нет прав
    end

    local status, err = pcall(function()
        -- Используем готовую функцию для получения точки прицела
        local basePos = GetAimPoint()

        if not basePos then
            return  -- Ничего не попало в луч, прерываем выполнение
        end

        local desiredBlockType = Player.CurrentBlock
        local desiredBlockForm = Player.CurrentKind

        -- ПРОВЕРКА: если тип блока равен 0 (воздух), ничего не делаем
        if desiredBlockType == 0 then
            -- Убираем вывод сообщения в чат
            hasShownNoBlockMessage = true
            return
        end

        local outOfBoundsCount = 0  -- счётчик блоков за границами
        local placedCount = 0       -- счётчик установленных блоков

        for x = 0, math.floor(blocksCountX) - 1 do
            for y = 0, math.floor(blocksCountY) - 1 do
                for z = 0, math.floor(blocksCountZ) - 1 do
                    local pos = CS.UnityEngine.Vector3(
                basePos.x + x,
                basePos.y + y,
                basePos.z + z
            )

            -- Проверка границ карты с использованием готовой функции
            if IsPositionInBounds(pos) then
                local currentBlockType = World:GetBlockType(pos)
                if currentBlockType ~= desiredBlockType then
                    World:AddBlock(pos, desiredBlockType, desiredBlockForm)
            placedCount = placedCount + 1
        end
    else

        end
    else

    end

local function getRange(startVal, endVal)
    local range = {}
    if startVal <= endVal then
        for val = startVal, endVal do
            table.insert(range, val)
        end
    else
        for val = startVal, endVal, -1 do
            table.insert(range, val)
        end
    end
    return range
end

-- Основная функция установки точки выделения
local function SetSelectionPoint(isFirstPoint)
    -- Проверка статического режима
    if staticSelectionMode then
        Chat:AddMessage("Для выделения новой области отключи статический режим отрисовки", "[f54949]")
        return false
    end

    -- Проверка прав администратора/модератора
    if not Player.IsAdmin and not Player.IsModerator then
        return false
    end

    local pos = GetAimPoint()
    if not pos then
        LogChatMessage("Не удалось определить позицию — проверьте видимость цели", "[f54949]")
        return false
    end

    if isFirstPoint then
        FirstPos = pos
        copied = false
        DisableBuildPreview()  -- Отключаем превью при установке первой точки
        Chat:AddMessage("Первая Точка Установлена: X= " .. pos.x .. ", Y= " .. pos.y .. ", Z= " .. pos.z, "[a455f2]")
    else
        SecondPos = pos
        copied = false
        Chat:AddMessage("Вторая Точка Установлена: X= " .. pos.x .. ", Y= " .. pos.y .. ", Z= " .. pos.z .. ", Область выделена", "[a455f2]")
    end

    return true
end

local function HandleSelectionKeyInput()
    -- Установка первой точки по клавише G
    if firstPosKey and CS.UnityEngine.Input.GetKeyDown(firstPosKey) then
        SetFirstPos()

function CopySelection()
    if not selectAreaMode then
        return
    end

    if not FirstPos or not SecondPos then
        Chat:AddMessage("Сначала выдели область.", "[f54949]")
        return
    end

    -- Если уже идёт копирование, блокируем повторный запуск
    if isCopying then
        Chat:AddMessage("Копирование уже выполняется!", "[f54949]")
        return
    end

    buildData.source = "clipboard"
    buildData.blocks = {}
    buildData.firstPos = FirstPos  -- сохраняем базовую точку выделения

    -- Определяем направление обхода для каждой оси
    local xStep = FirstPos.x <= SecondPos.x and 1 or -1
    local yStep = FirstPos.y <= SecondPos.y and 1 or -1
    local zStep = FirstPos.z <= SecondPos.z and 1 or -1

    local outOfBoundsCount = 0  -- счётчик позиций за границами
    local airBlocksSkipped = 0  -- счётчик пропущенных блоков воздуха

    -- Копируем блоки в порядке от FirstPos к SecondPos
    for x = FirstPos.x, SecondPos.x, xStep do
        for y = FirstPos.y, SecondPos.y, yStep do
            for z = FirstPos.z, SecondPos.z, zStep do
                local pos = CS.UnityEngine.Vector3(x, y, z)

                -- ПРОВЕРКА: находится ли позиция в пределах карты
                if IsPositionInBounds(pos) then
                    -- Получаем данные о блоке
                    local blockType = World:GetBlockType(pos)

            -- ФИЛЬТРАЦИЯ: сохраняем только реальные блоки (blockType ≠ 0)
            if blockType ~= 0 then
                -- Рассчитываем offset относительно FirstPos (базовой точки выделения)
                local offsetX = x - FirstPos.x
                local offsetY = y - FirstPos.y
                local offsetZ = z - FirstPos.z

                local blockForm = World:GetBlockKind(pos) or 0

                -- Сохраняем в buildData.blocks
                table.insert(buildData.blocks, {
                    offset = CS.UnityEngine.Vector3(offsetX, offsetY, offsetZ),
            blockType = blockType,
            blockForm = blockForm
        })
            else
                airBlocksSkipped = airBlocksSkipped + 1
            end
        else
            outOfBoundsCount = outOfBoundsCount + 1
        end
            end
        end
    end
    Chat:AddMessage("Область скопирована.", "[a455f2]")

    -- СБРАСЫВАЕМ флаг копирования
    isCopying = false
end


local function SerializeBlocksToJson(blocks)
    local parts = {}
    table.insert(parts, '{"blocks":[')
    for i, block in ipairs(blocks) do
        local part = string.format(
            '{"offset":{"x":%d,"y":%d,"z":%d},"blockType":%d,"blockForm":%d}',
            block.offset.x, block.offset.y, block.offset.z,
            block.blockType, block.blockForm
        )
        table.insert(parts, part)
        if i < #blocks then
            table.insert(parts, ',')
        end
    end
    table.insert(parts, ']}')
    return table.concat(parts)
end

-- Кодируем все не ASCII символы в %XX формат
local function encode_filename(str)
    return (str:gsub("[^%w%.%-]", function(c)
        return string.format("%%%02X", string.byte(c))
    end))
end

-- Декодируем обратно %XX в символы
local function decode_filename(str)
    return (str:gsub("%%(%x%x)", function(h)
        return string.char(tonumber(h,16))
    end))
end

local function containsRussian(str)
    return str:find("[\208-\209]") ~= nil -- приблизительная проверка на кириллицу в UTF-8
end

function SaveSelectionToFile(filename)
    if not FirstPos or not SecondPos then
       Chat:AddMessage("Сначала скопируй область", "[f54949]")
        return
    end

    CreateBuildsFolderIfNotExists()

    local blocks = {}
    local minX = math.min(FirstPos.x, SecondPos.x)
    local maxX = math.max(FirstPos.x, SecondPos.x)
    local minY = math.min(FirstPos.y, SecondPos.y)
    local maxY = math.max(FirstPos.y, SecondPos.y)
    local minZ = math.min(FirstPos.z, SecondPos.z)
    local maxZ = math.max(FirstPos.z, SecondPos.z)

    for x = minX, maxX do
        for y = minY, maxY do
            for z = minZ, maxZ do
                local pos = CS.UnityEngine.Vector3(x, y, z)
                local blockType = World:GetBlockType(pos)
                if blockType ~= 0 then
                    local blockForm = 0
                    if World.GetBlockKind then
                        blockForm = World:GetBlockKind(pos)
                    end
                    local offset = {x = x - minX, y = y - minY, z = z - minZ}
                    table.insert(blocks, {offset = offset, blockType = blockType, blockForm = blockForm})
                end
            end
        end
    end

    local jsonStr = SerializeBlocksToJson(blocks)
    local encodedFilename = encode_filename(filename)

    local filepath = buildsFolder .. encodedFilename
    local file = io.open(filepath, "w")
    if file then
        file:write(jsonStr)
        file:close()
       Chat:AddMessage("Область успешно сохранена: " .. filename, "[a455f2]")
        UpdateSavedBuildsList()
    else
       LogChatMessage("Ошибка записи файла: " .. filename, "[f54949]")
    end
end

    CalculatePreviewBounds()

    useCopiedBlocks = false
    Chat:AddMessage("Файл " .. decodedFilename .. " загружен. Всего блоков: " .. tostring(#buildData.blocks), "[a455f2]")

    -- После загрузки новой постройки сбрасываем меш, чтобы он пересоздался
    CleanupPreviewMesh()
end

local function CollectBlocksInSelection()
    if not FirstPos or not SecondPos then
        Chat:AddMessage("Сначала выдели область", "[f54949]")
        return {}
    end

    local xOrder = getRange(FirstPos.x, SecondPos.x)
    local yOrder = getRange(FirstPos.y, SecondPos.y)
    local zOrder = getRange(FirstPos.z, SecondPos.z)

    local blocks = {}

    for _, x in ipairs(xOrder) do
        for _, z in ipairs(zOrder) do
            for _, y in ipairs(yOrder) do
                local pos = CS.UnityEngine.Vector3(x, y, z)
                local blockType = World:GetBlockType(pos)
                if blockType ~= 0 then
                    table.insert(blocks, pos)
                end
            end
        end
    end

    return blocks
end


local function CollectAllPositionsInSelection()
    if not FirstPos or not SecondPos then
       -- Chat:AddMessage("Сначала выдели область", "[f54949]")
        return {}
    end

    local xOrder = getRange(FirstPos.x, SecondPos.x)
    local yOrder = getRange(FirstPos.y, SecondPos.y)
    local zOrder = getRange(FirstPos.z, SecondPos.z)

    local positions = {}

    for _, x in ipairs(xOrder) do
        for _, z in ipairs(zOrder) do
            for _, y in ipairs(yOrder) do
                local pos = CS.UnityEngine.Vector3(x, y, z)
                table.insert(positions, pos)
            end
        end
    end

    return positions
end

function OpenFolder(path)
    -- Нормализуем путь: заменяем обратные слеши на прямые
    path = path:gsub("\\", "/")

    -- Проверяем существование папки
    local exists = CS.System.IO.Directory.Exists(path)
    if not exists then
        LogChatMessage("Папка не существует: " .. path, "[ff0000]")
        -- Пытаемся создать папку
        local success, err = pcall(function()
            CS.System.IO.Directory.CreateDirectory(path)
        end)
        if success then
            LogChatMessage("Папка создана: " .. path, "[00ff00]")
            exists = true
        else
            LogChatMessage("Ошибка создания папки: " .. (err or "unknown"), "[ff0000]")
            return
        end
    end

    -- Отладка: выводим путь в чат
    print("Открываем папку: " .. path)
    LogChatMessage("Открываем папку: " .. path, "[ffff00]")

    -- Формируем корректный URL для OpenURL
    local folderPath
    if CS.UnityEngine.Application.platform == CS.UnityEngine.RuntimePlatform.WindowsPlayer or
       CS.UnityEngine.Application.platform == CS.UnityEngine.RuntimePlatform.WindowsEditor then
        -- Для Windows: используем file:/// с тремя слешами
        folderPath = "file:///" .. path
    else
        -- Для macOS и Linux: используем file:// с двумя слешами
        folderPath = "file://" .. path
    end

    -- Открываем папку через OpenURL
    CS.UnityEngine.Application.OpenURL(folderPath)
end

function SplitIntoChunks(blockList, chunkSize)
    local chunks = {}
    for i = 1, #blockList, chunkSize do
        local chunk = {}
        for j = i, math.min(i + chunkSize - 1, #blockList) do
            table.insert(chunk, blockList[j])
        end
        table.insert(chunks, chunk)
    end
    return chunks
end

function StartFilling()
    if not selectAreaMode then
        return
    end

    local desiredBlockType = Player.CurrentBlock
    local desiredBlockForm = Player.CurrentKind

    fillPositions = CollectAllPositionsInSelection()
    if #fillPositions == 0 then
        Chat:AddMessage("Сначала выдели область", "[f54949]")
        return
    end

    -- Переиспользуем lastActionBlocks
    if not lastActionBlocks then
        lastActionBlocks = {}
    else
        for k in pairs(lastActionBlocks) do lastActionBlocks[k] = nil end
    end

    -- Переиспользуем blocksToReplace
    if not blocksToReplace then
        blocksToReplace = {}
    else
        for k in pairs(blocksToReplace) do blocksToReplace[k] = nil end
    end

    local fillIndex = 1  -- используем индекс вместо table.insert

    for _, pos in ipairs(fillPositions) do
        local oldBlockType = World:GetBlockType(pos)
        local oldBlockForm = World:GetBlockKind(pos)

        -- Если блок уже нужный — пропускаем
        if oldBlockType == desiredBlockType and oldBlockForm == desiredBlockForm then
            goto continue
        end

        -- Сохраняем старое состояние ТОЛЬКО для изменённых блоков
        lastActionBlocks[fillIndex] = {
            pos = pos,
            blockType = oldBlockType,
            blockForm = oldBlockForm
        }

        blocksToReplace[fillIndex] = {
            pos = pos,
            blockType = desiredBlockType,
            blockForm = desiredBlockForm,
            processed = false
        }
        fillIndex = fillIndex + 1

        ::continue::
    end

    currentFillIndex = 1
    isFilling = true
    isReplacing = false
    isDeleting = false
    isPlacing = false

    LogChatMessage(
        string.format("Заполнение: %d блоков требуют изменения", (fillIndex - 1)),
        "[9f40ff]"
    )
end

function StartDestroy()
    if not selectAreaMode then
        Chat:AddMessage("Включи режим копирования для работы с областью.", "[f54949]")
        return
    end

    if isDeleting then  -- защита от повторного запуска
        Chat:AddMessage("Удаление уже выполняется!", "[f54949]")
        return
    end

    local rawBlocks = CollectBlocksInSelection()
    if #rawBlocks == 0 then
        Chat:AddMessage("В выделенной области нет блоков для удаления.", "[f54949]")
        return
    end

    -- Определяем целевой тип блока, если режим "selected_type"
    local targetBlockType = nil
    if deleteMode == "selected_type" then
        targetBlockType = Player.CurrentBlock
        if not targetBlockType then
            Chat:AddMessage("ОШИБКА: Текущий блок не выбран в инвентаре", "[f54949]")
            return
        end
    end

    -- КЭШИРОВАНИЕ: получаем все данные о блоках ОДИН РАЗ
    -- Переиспользуем cachedBlocks, если она уже существует
    if not cachedBlocks then
        cachedBlocks = {}
    else
        for k in pairs(cachedBlocks) do cachedBlocks[k] = nil end
    end

    for _, pos in ipairs(rawBlocks) do
        cachedBlocks[pos] = {
            type = World:GetBlockType(pos),
            form = World:GetBlockKind(pos) or 0
        }
    end

    -- Сохраняем текущее состояние блоков для Undo
    -- ПЕРЕИНИЦИАЛИЗАЦИЯ lastActionBlocks (вместо переиспользования)
    lastActionBlocks = {}  -- создаём новую таблицу

    -- Инициализируем области для пересчёта освещения
    minX, minY, minZ = math.huge, math.huge, math.huge
    maxX, maxY, maxZ = -math.huge, -math.huge, -math.huge

    -- Переиспользуем blocksToDelete
    if not blocksToDelete then
        blocksToDelete = {}
    else
        for k in pairs(blocksToDelete) do blocksToDelete[k] = nil end
    end

    local deleteIndex = 1  -- используем индекс вместо table.insert

    for _, pos in ipairs(rawBlocks) do
        local blockData = cachedBlocks[pos]
        local oldType = blockData.type
        local oldForm = blockData.form

        -- Пропускаем блок, если режим "selected_type" и тип не совпадает
        if deleteMode == "selected_type" and oldType ~= targetBlockType then
            goto continue
        end

        -- Заполняем по индексу вместо table.insert
        lastActionBlocks[deleteIndex] = {pos = pos, blockType = oldType, blockForm = oldForm}

        -- Обновляем границы области
        if pos.x < minX then minX = pos.x end
        if pos.y < minY then minY = pos.y end
        if pos.z < minZ then minZ = pos.z end

        if pos.x > maxX then maxX = pos.x end
        if pos.y > maxY then maxY = pos.y end
        if pos.z > maxZ then maxZ = pos.z end

        -- Добавляем блок в список удаления с флагом обработки
        blocksToDelete[deleteIndex] = {
            pos = pos,
            processed = false
        }
        deleteIndex = deleteIndex + 1

        ::continue::
    end

    if deleteIndex == 1 then  -- deleteIndex не увеличился, значит, блоков для удаления нет
        if deleteMode == "selected_type" then
            Chat:AddMessage("В выделенной области нет блоков типа " .. targetBlockType, "[f54949]")
        else
            Chat:AddMessage("В выделенной области нет блоков для удаления.", "[f54949]")
        end
        return
    end

    currentDeleteIndex = 1
    isDeleting = true
    isReplacing = false
    isPlacing = false
    isFilling = false

    Chat:AddMessage("Запуск удаления. Режим: " ..
        (deleteMode == "all" and "все блоки" or "только выбранный тип") ..
        ". Блоков: " .. (deleteIndex - 1), "[a455f2]")
end

function StartReplacing()
    if not selectAreaMode then
        return
    end

    local desiredBlockType = Player.CurrentBlock
    local desiredBlockForm = Player.CurrentKind

    -- Получаем все позиции в выделенной области
    local blocks = CollectAllPositionsInSelection()
    if #blocks == 0 then
        Chat:AddMessage("Область пуста или не выделена", "[f54949]")
        return
    end

    -- Переиспользуем lastActionBlocks
    if not lastActionBlocks then
        lastActionBlocks = {}
    else
        for k in pairs(lastActionBlocks) do lastActionBlocks[k] = nil end
    end

    -- Переиспользуем blocksToReplace
    if not blocksToReplace then
        blocksToReplace = {}
    else
        for k in pairs(blocksToReplace) do blocksToReplace[k] = nil end
    end

    local replaceIndex = 1  -- используем индекс вместо table.insert

    for _, pos in ipairs(blocks) do
        -- Проверяем, находится ли позиция в границах карты
        if not IsPositionInBounds(pos) then
            goto continue
        end

        local oldBlockType = World:GetBlockType(pos)
        local oldBlockForm = World:GetBlockKind(pos)

        -- ПРОПУСКАЕМ ВОЗДУШНЫЕ БЛОКИ (тип 0)
        if oldBlockType == 0 then  -- 0 обычно означает воздух/пустое пространство
            goto continue
        end

        -- Определяем, нужно ли заменять блок в зависимости от режима
        local shouldReplace = false

        if replaceMode == "texture" then
            -- Только текстура: заменяем, если тип отличается
            if oldBlockType ~= desiredBlockType then
                shouldReplace = true
            end
        else  -- replaceMode == "full"
            -- Полностью: заменяем, если тип ИЛИ форма отличаются
            if oldBlockType ~= desiredBlockType or oldBlockForm ~= desiredBlockForm then
                shouldReplace = true
            end
        end

        if not shouldReplace then
            goto continue
        end

        -- Сохраняем старое состояние для возможности отмены
        lastActionBlocks[replaceIndex] = {
            pos = pos,
            blockType = oldBlockType,
            blockForm = oldBlockForm
        }

        blocksToReplace[replaceIndex] = {
            pos = pos,
            blockType = desiredBlockType,
            blockForm = replaceMode == "texture" and oldBlockForm or desiredBlockForm,
            processed = false
        }
        replaceIndex = replaceIndex + 1

        ::continue::
    end

    currentReplaceIndex = 1
    isReplacing = true
    isDeleting = false
    isPlacing = false
    isFilling = false

    Chat:AddMessage("Запуск замены. Режим: " .. (replaceMode == "texture" and "только текстура" or "всё (тип+форма)") .. ". Блоков для замены: " .. (replaceIndex - 1), "[a455f2]")
end

function IsPositionInBounds(pos)
    if World.IsInfinite then
        return true  -- На бесконечной карте все позиции допустимы
    end
    -- Для ограниченной карты проверяем строго внутри границ
    return pos.x >= 0 and pos.x < World.SizeX and
           pos.y >= 0 and pos.y < World.SizeY and
           pos.z >= 0 and pos.z < World.SizeZ
end

function CheckMapBounds()
    if World.IsInfinite then
       -- Chat:AddMessage("Карта бесконечная — ограничений нет", "[a455f2]")
    else
        LogChatMessage(
            string.format("Границы карты: X[0-%d], Y[0-%d], Z[0-%d]",
            World.SizeX - 1, World.SizeY - 1, World.SizeZ - 1),
            "[a455f2]"
        )
    end
end

function StartPasting()
    if not selectAreaMode then
        return
    end

    -- Блокировка: если идёт копирование или вставка, не разрешаем начать новую вставку
    if isCopying then
        Chat:AddMessage("Нельзя вставить область: копирование ещё не завершено!", "[f54949]")
        return
    end

    if isPasteInProgress then
        Chat:AddMessage("Постройка уже выполняется! Дождись завершения.", "[f54949]")
        return
    end

    -- Используем глобальную buildData, не создаём локальную
    if not buildData.source or #buildData.blocks == 0 then
        Chat:AddMessage("Нет данных для постройки!!!", "[f54949]")
        return
    end

    local pos = GetAimPoint()
    if not pos then
        Chat:AddMessage("Не удалось определить позицию для постройки", "[f54949]")
        return
    end

    isPasteInProgress = true  -- устанавливаем флаг: вставка началась

    -- Очистка данных перед использованием
    blocksToPlace = blocksToPlace or {}
    for k in pairs(blocksToPlace) do blocksToPlace[k] = nil end

    lastActionBlocks = lastActionBlocks or {}
    for k in pairs(lastActionBlocks) do lastActionBlocks[k] = nil end

    local skippedCount = 0
    local outOfBoundsCount = 0
    local alreadyExistsCount = 0

    local placeIndex = 1

    for _, data in ipairs(buildData.blocks) do
        local worldPos = CS.UnityEngine.Vector3(
            pos.x + data.offset.x,
            pos.y + data.offset.y,
            pos.z + data.offset.z
        )

        if IsPositionInBounds(worldPos) then
            -- Читаем данные о блоке только один раз
            local existingBlockType = World:GetBlockType(worldPos)
            local existingBlockForm = 0

            -- ПОЛУЧАЕМ ФОРМУ ТОЛЬКО ЕСЛИ БЛОК СУЩЕСТВУЕТ (blockType ≠ 0)
            if existingBlockType ~= 0 then
                existingBlockForm = World:GetBlockKind(worldPos) or 0
            end

            -- Сохраняем исходное состояние блока для отмены
            lastActionBlocks[placeIndex] = {
                pos = worldPos,
                blockType = existingBlockType,
                blockForm = existingBlockForm
            }

            if existingBlockType ~= 0 and
               existingBlockType == data.blockType and
               existingBlockForm == data.blockForm then
                alreadyExistsCount = alreadyExistsCount + 1
            else
                blocksToPlace[placeIndex] = {
                    pos = worldPos,
            blockType = data.blockType,
            blockForm = data.blockForm,
            processed = false  -- флаг обработки
        }
                placeIndex = placeIndex + 1
            end
        else
            outOfBoundsCount = outOfBoundsCount + 1
        end
    end

    -- Сообщения о прогрессе
    if outOfBoundsCount > 0 then
        LogChatMessage(
            "Пропущено " .. outOfBoundsCount .. " блоков за границами карты",
            "[f54949]"
        )
    end

    if alreadyExistsCount > 0 then
        LogChatMessage(
            "Пропущено " .. alreadyExistsCount .. " уже существующих блоков",
            "[ffff00]"
        )
    end

    if placeIndex == 1 then  -- placeIndex не увеличился, значит, ставить нечего
        isPasteInProgress = false  -- сброс флага, если нечего ставить
        LogChatMessage(
            "Все блоки уже стоят в этих позициях.",
            "[f54949]"
        )
        return
    end

    currentPlaceIndex = 1
    isPlacing = true
    isDeleting = false
    isReplacing = false

    Chat:AddMessage("Запуск постройки. Блоков для установки: " .. (placeIndex - 1), "[a455f2]")
end

function CalculatePreviewBounds()
    if not previewBuildData or #previewBuildData == 0 then
        previewFirstPos = nil
        previewSecondPos = nil
        return
    end

    local minX, minY, minZ = math.huge, math.huge, math.huge
    local maxX, maxY, maxZ = -math.huge, -math.huge, -math.huge

    for _, data in ipairs(previewBuildData) do
        local x, y, z = data.offset.x, data.offset.y, data.offset.z
        minX = math.min(minX, x)
        minY = math.min(minY, y)
        minZ = math.min(minZ, z)
        maxX = math.max(maxX, x)
        maxY = math.max(maxY, y)
        maxZ = math.max(maxZ, z)
    end

    previewFirstPos = CS.UnityEngine.Vector3(minX, minY, minZ)
    previewSecondPos = CS.UnityEngine.Vector3(maxX, maxY, maxZ)  -- +1 для корректного отображения границ
end

function DrawSelectionBoxGL()
    if not isSelectionVisible then
        return
    end

    if not selectAreaMode then  -- если режим выделения выключен, не отрисовываем
        return
    end

    if not Player.IsAdmin and not Player.IsModerator then
        return
    end

    if not FirstPos or not SecondPos then
        return
    end

    local cam = CS.UnityEngine.Camera.current
    if not cam then
        return
    end

    -- Инициализация материала (существующий код)
    if not crossMaterial then
        crossMaterial = CS.UnityEngine.Material(CS.UnityEngine.Shader.Find("Sprites/Default"))
        if not crossMaterial then
            return
        end
        crossMaterial:SetInt("_SrcBlend", CS.UnityEngine.Rendering.BlendMode.SrcAlpha)
        crossMaterial:SetInt("_DstBlend", CS.UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha)
        crossMaterial:SetInt("_Cull", CS.UnityEngine.Rendering.CullMode.Off)
        crossMaterial:SetInt("_ZWrite", 0)
        crossMaterial:SetInt("_ZTest", 0)
        crossMaterial:EnableKeyword("_ALPHABLEND_ON")
    end

    local passSuccess = crossMaterial:SetPass(0)
    if not passSuccess then
        return
    end

    -- Определяем позиции для отрисовки
    local drawFirstPos, drawSecondPos, drawRotation

    if staticSelectionMode and staticFirstPos and staticSecondPos then
        -- В статическом режиме используем изначальные координаты без поворота
        drawFirstPos = staticFirstPos
        drawSecondPos = staticSecondPos
        drawRotation = 0
    else
        -- В динамическом режиме используем текущие позиции и поворот
        drawFirstPos = FirstPos + offset
        drawSecondPos = SecondPos + offset
        drawRotation = currentRotation
    end

    -- Расчёт границ с учётом поворота
    local rotatedFirstPos = drawFirstPos
    local rotatedSecondPos = RotatePointAroundY(drawSecondPos, drawFirstPos, -drawRotation)

    local minX = math.min(rotatedFirstPos.x, rotatedSecondPos.x)
    local maxX = math.max(rotatedFirstPos.x, rotatedSecondPos.x) + 1
    local minY = math.min(rotatedFirstPos.y, rotatedSecondPos.y)
    local maxY = math.max(rotatedFirstPos.y, rotatedSecondPos.y) + 1
    local minZ = math.min(rotatedFirstPos.z, rotatedSecondPos.z)
    local maxZ = math.max(rotatedFirstPos.z, rotatedSecondPos.z) + 1

    -- Создание углов коробки
    local corners = {
        CS.UnityEngine.Vector3(minX, minY, minZ),
        CS.UnityEngine.Vector3(maxX, minY, minZ),
        CS.UnityEngine.Vector3(maxX, maxY, minZ),
        CS.UnityEngine.Vector3(minX, maxY, minZ),
        CS.UnityEngine.Vector3(minX, minY, maxZ),
        CS.UnityEngine.Vector3(maxX, minY, maxZ),
        CS.UnityEngine.Vector3(maxX, maxY, maxZ),
        CS.UnityEngine.Vector3(minX, maxY, maxZ)
    }

    -- Отрисовка (существующий код GL)
    CS.UnityEngine.GL.PushMatrix()
    CS.UnityEngine.GL.LoadProjectionMatrix(cam.projectionMatrix)
    CS.UnityEngine.GL.modelview = cam.worldToCameraMatrix
    
    -- Смещение граней вперёд на 0.01 единицы по глубине
local offsetZ = -0.01
for i = 1, 8 do
    corners[i] = corners[i] + cam.transform.forward * offsetZ
end

    -- === ОТРИСОВКА ПОЛУПРОЗРАЧНЫХ ГРАНЕЙ (как в превью) ===
    CS.UnityEngine.GL.Begin(CS.UnityEngine.GL.TRIANGLES)
    -- Полупрозрачный серый цвет для граней, прозрачность 0.2
    CS.UnityEngine.GL.Color(CS.UnityEngine.Color(0, 0, 0, 0.2))

    -- НИЖНИЙ КВАДРАТ (Z = minZ) — дно
    CS.UnityEngine.GL.Vertex(corners[1])
    CS.UnityEngine.GL.Vertex(corners[2])
    CS.UnityEngine.GL.Vertex(corners[3])

    CS.UnityEngine.GL.Vertex(corners[1])
    CS.UnityEngine.GL.Vertex(corners[3])
    CS.UnityEngine.GL.Vertex(corners[4])

    -- ВЕРХНИЙ КВАДРАТ (Z = maxZ) — потолок

function InitializePreviewResources()
    if isMeshInitialized then return end

    -- Пытаемся получить встроенный меш куба
    blockMesh = CS.UnityEngine.Resources.GetBuiltinResource(
        typeof(CS.UnityEngine.Mesh),
        "Cube"
    )

    -- Если не получилось, создаём временный куб (один раз!)
    if not blockMesh then
        local cube = CS.UnityEngine.GameObject.CreatePrimitive(CS.UnityEngine.PrimitiveType.Cube)
        local meshFilter = cube:GetComponent(typeof(CS.UnityEngine.MeshFilter))
        if meshFilter then
            blockMesh = meshFilter.mesh
        end
        CS.UnityEngine.Object.DestroyImmediate(cube)
    end

    isMeshInitialized = true
end

function CreateCombinedBlockMeshChunks(blockDataList, maxBlocksPerChunk)
    maxBlocksPerChunk = maxBlocksPerChunk or 1000
    local chunks = {}

    for i = 1, #blockDataList, maxBlocksPerChunk do
        local chunkBlocks = {}
        local endIndex = math.min(i + maxBlocksPerChunk - 1, #blockDataList)

        for j = i, endIndex do
            -- Сохраняем смещение относительно previewFirstPos (угла области)
            local localOffset = blockDataList[j].offset - previewFirstPos
            table.insert(chunkBlocks, {
                offset = localOffset,
                type = blockDataList[j].type  -- если нужно
            })
        end

        local chunkMesh = CreateSingleChunkMesh(chunkBlocks)
        if chunkMesh then
            table.insert(chunks, {
                mesh = chunkMesh,
                offset = CS.UnityEngine.Vector3.zero  -- будет обновляться при повороте
            })
        end
    end

    print("Created " .. #chunks .. " chunks from " .. #blockDataList .. " blocks")
    return chunks
end

function CreateSingleChunkMesh(blockDataList)
    if not blockDataList or #blockDataList == 0 then return nil end

    local vertices = {}
    local triangles = {}

    -- Базовый шаблон куба (8 вершин)
    local cubeVertices = {
        CS.UnityEngine.Vector3(0, 0, 0),  -- 0
        CS.UnityEngine.Vector3(1, 0, 0),  -- 1
        CS.UnityEngine.Vector3(1, 1, 0),  -- 2
        CS.UnityEngine.Vector3(0, 1, 0),  -- 3
        CS.UnityEngine.Vector3(0, 0, 1),  -- 4
        CS.UnityEngine.Vector3(1, 0, 1),  -- 5
        CS.UnityEngine.Vector3(1, 1, 1),  -- 6
        CS.UnityEngine.Vector3(0, 1, 1)   -- 7
    }

    -- Индексы треугольников для всех 6 граней (порядок против часовой стрелки)
    local cubeTriangles = {
        -- Передняя грань (Z = 0)
        0, 3, 2,
        0, 2, 1,

        -- Задняя грань (Z = 1)
        5, 6, 7,
        5, 7, 4,

        -- Левая грань (X = 0)
        4, 7, 3,
        4, 3, 0,

        -- Правая грань (X = 1)
        1, 2, 6,
        1, 6, 5,

        -- Верхняя грань (Y = 1)
        2, 3, 7,
        2, 7, 6,

        -- Нижняя грань (Y = 0)
        4, 5, 1,
        4, 1, 0
    }

    local vertexOffset = 0

    for _, blockData in ipairs(blockDataList) do
        if blockData and blockData.offset then
            local x, y, z = blockData.offset.x, blockData.offset.y, blockData.offset.z

            -- Добавляем вершины для этого блока (смещённые)
            for _, v in ipairs(cubeVertices) do
                table.insert(vertices, CS.UnityEngine.Vector3(v.x + x, v.y + y, v.z + z))
            end

            -- Добавляем индексы треугольников (с учётом смещения)
            for i = 1, #cubeTriangles do
                table.insert(triangles, cubeTriangles[i] + vertexOffset)
            end

            vertexOffset = vertexOffset + 8
        else
            print("Invalid block data found, skipping")
        end
    end

    -- Создаём меш
    local mesh = CS.UnityEngine.Mesh()
    mesh.vertices = vertices
    mesh.triangles = triangles
    mesh:RecalculateNormals()

    return mesh
end

function InitializePreviewMesh()
    if isMeshInitialized or not previewBuildData or #previewBuildData == 0 then
        return
    end

    combinedMeshChunks = CreateCombinedBlockMeshChunks(previewBuildData, 1000)

    if combinedMeshChunks and #combinedMeshChunks > 0 then
        isMeshInitialized = true

        selectionMaterial = CS.UnityEngine.Material(CS.UnityEngine.Shader.Find("Standard"))
        if selectionMaterial then
            -- Основной цвет с прозрачностью (альфа = 0.4)
            selectionMaterial.color = CS.UnityEngine.Color(0.28, 0.28, 0.28, 0.40)

            -- Металлик: 0 — полностью неметаллический (пластик)
            selectionMaterial:SetFloat("_Metallic", 0)

            -- Гладкость: 0.1 — слегка матовая поверхность
            selectionMaterial:SetFloat("_Glossiness", 0)

            -- Режим рендеринга: прозрачный
            selectionMaterial.renderQueue = CS.UnityEngine.Rendering.RenderQueue.Transparent

            -- Включение альфа‑смешивания
            selectionMaterial:EnableKeyword("_ALPHABLEND_ON")

            -- Отключение записи в Z‑буфер (КРИТИЧНО для прозрачности)
            selectionMaterial:SetInt("_ZWrite", 0)

            -- Отсечение граней: отображаем обе стороны
            selectionMaterial:SetInt("_Cull", CS.UnityEngine.Rendering.CullMode.Off)
        end

        -- МАТЕРИАЛ ДЛЯ ОБВОДКИ (чёрный, непрозрачный)
        if not crossMaterial then
            crossMaterial = CS.UnityEngine.Material(CS.UnityEngine.Shader.Find("Sprites/Default"))
            if crossMaterial then
                crossMaterial:SetInt("_SrcBlend", CS.UnityEngine.Rendering.BlendMode.SrcAlpha)
                crossMaterial:SetInt("_DstBlend", CS.UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha)
                crossMaterial:SetInt("_Cull", CS.UnityEngine.Rendering.CullMode.Off)
                crossMaterial:SetInt("_ZWrite", 0)
                crossMaterial:EnableKeyword("_ALPHABLEND_ON")
            end
        end
    end
end

function UpdatePreviewMeshPosition(newAimPos, rotationAngle)
    if not isMeshInitialized then return end


    -- Сохраняем базовую позицию области
    previewMeshTransform = newAimPos

    -- Создаём кватернион поворота вокруг оси Y
    local rotationQuaternion = CS.UnityEngine.Quaternion.Euler(0, rotationAngle, 0)

    -- Сохраняем поворот для использования при отрисовке
    previewMeshRotation = rotationQuaternion

    -- Если угол поворота не равен 0, скрываем меш
    showMeshWhenRotated = (rotationAngle == 0)
end

function CleanupPreviewMesh()
    if combinedMeshChunks then
        for _, chunk in ipairs(combinedMeshChunks) do
            if chunk.mesh then
                CS.UnityEngine.Object.DestroyImmediate(chunk.mesh)
            end
        end
        combinedMeshChunks = nil
    end
    isMeshInitialized = false
    previewMeshTransform = CS.UnityEngine.Vector3.zero
    print("Preview mesh cleaned up")
end

function DrawBuildPreviewGL()
    if not isPreviewVisible then return end
    if not showBuildPreview or not previewFirstPos or not previewSecondPos then return end

    local cam = CS.UnityEngine.Camera.current
    if not cam then return end

    InitializePreviewMesh()

    if not selectionMaterial then
        print("ERROR: selectionMaterial not initialized")
        return
    end

    local passSuccess = selectionMaterial:SetPass(0)
    if not passSuccess then return end

    local aimPos = GetAimPoint()
    if not aimPos then return end

    local drawFirstPos = aimPos + previewFirstPos
    local drawSecondPos = aimPos + previewSecondPos
    local drawRotation = currentRotation

    local rotatedFirstPos = drawFirstPos
    local rotatedSecondPos = RotatePointAroundY(drawSecondPos, drawFirstPos, -drawRotation)

    local minX = math.min(rotatedFirstPos.x, rotatedSecondPos.x)
    local maxX = math.max(rotatedFirstPos.x, rotatedSecondPos.x) + 1
    local minY = math.min(rotatedFirstPos.y, rotatedSecondPos.y)
    local maxY = math.max(rotatedFirstPos.y, rotatedSecondPos.y) + 1
    local minZ = math.min(rotatedFirstPos.z, rotatedSecondPos.z)
    local maxZ = math.max(rotatedFirstPos.z, rotatedSecondPos.z) + 1

    local corners = {
        CS.UnityEngine.Vector3(minX, minY, minZ),
        CS.UnityEngine.Vector3(maxX, minY, minZ),
        CS.UnityEngine.Vector3(maxX, maxY, minZ),
        CS.UnityEngine.Vector3(minX, maxY, minZ),
        CS.UnityEngine.Vector3(minX, minY, maxZ),
        CS.UnityEngine.Vector3(maxX, minY, maxZ),
        CS.UnityEngine.Vector3(maxX, maxY, maxZ),
        CS.UnityEngine.Vector3(minX, maxY, maxZ)
    }

    CS.UnityEngine.GL.PushMatrix()
    CS.UnityEngine.GL.LoadProjectionMatrix(cam.projectionMatrix)
    CS.UnityEngine.GL.modelview = cam.worldToCameraMatrix

    -- Смещение граней вперёд на 0.01 единицы по глубине
    local offsetZ = 0.01
    for i = 1, 8 do
        corners[i] = corners[i] + cam.transform.forward * offsetZ
    end

    -- ОТРИСОВКА МЕША (только если угол поворота равен 0)
if previewBuildData and #previewBuildData > 0 and combinedMeshChunks and showMeshWhenRotated then
    UpdatePreviewMeshPosition(aimPos, currentRotation)

    if isMeshInitialized then
        if selectionMaterial and selectionMaterial:SetPass(0) then
            for _, chunk in ipairs(combinedMeshChunks) do
                local worldPosition = aimPos + chunk.offset
                local matrix = CS.UnityEngine.Matrix4x4.TRS(
                    worldPosition,
            previewMeshRotation,
            CS.UnityEngine.Vector3.one
        )
        CS.UnityEngine.Graphics.DrawMesh(
            chunk.mesh,
            matrix,
            selectionMaterial,
            0,
            nil,
            0
        )
            end
        end
    end
else
    -- При повороте очищаем меш, но не удаляем ресурсы — только скрываем
    -- CleanupPreviewMesh() не вызываем, чтобы не терять данные
end

    -- ОТРИСОВКА КАРКАСА КОРОБКИ (только линии, без граней)
    if crossMaterial and crossMaterial:SetPass(0) then
        CS.UnityEngine.GL.Begin(CS.UnityEngine.GL.LINES)
        CS.UnityEngine.GL.Color(CS.UnityEngine.Color(0.2, 0.2, 0.2, 1))

        -- Нижний квадрат
        for _, pair in ipairs({{1,2},{2,3},{3,4},{4,1}}) do
            CS.UnityEngine.GL.Vertex(corners[pair[1]])
            CS.UnityEngine.GL.Vertex(corners[pair[2]])
        end

        -- Верхний квадрат
        for _, pair in ipairs({{5,6},{6,7},{7,8},{8,5}}) do
            CS.UnityEngine.GL.Vertex(corners[pair[1]])
            CS.UnityEngine.GL.Vertex(corners[pair[2]])
        end

        -- Вертикальные линии
        for i=1,4 do
            CS.UnityEngine.GL.Vertex(corners[i])
            CS.UnityEngine.GL.Vertex(corners[i+4])
        end

        CS.UnityEngine.GL.End()
    else
        print("WARNING: crossMaterial not available for outline drawing")
    end

    CS.UnityEngine.GL.PopMatrix()
end

function ToggleStaticSelectionMode()
    staticSelectionMode = not staticSelectionMode

    if staticSelectionMode then
        -- Сохраняем изначальные координаты и угол поворота
        if FirstPos and SecondPos then
            originalFirstPos = FirstPos
            originalSecondPos = SecondPos
            originalRotation = currentRotation

            -- Фиксируем область в изначальных координатах без поворота
            staticFirstPos = originalFirstPos
            staticSecondPos = originalSecondPos
            currentRotation = 0  -- сбрасываем поворот для статического режима

            Chat:AddMessage("Активирован статический режим. Сменить режим - [ " .. GetKeyName(areaModedKey) .. " ]", "[a455f2]")
        else
            --Chat:AddMessage("Не удалось активировать статический режим: область не выделена.", "[f54949]")
            staticSelectionMode = false
        end
    else
        -- При выключении статического режима восстанавливаем последний поворот
        if originalFirstPos and originalSecondPos then
            FirstPos = originalFirstPos
            SecondPos = originalSecondPos
            currentRotation = originalRotation
        end
        Chat:AddMessage("Активирован динамический режим. Сменить режим - [ " .. GetKeyName(areaModedKey) .. " ]", "[a455f2]")
    end
end

function HexToColor(hex)
    -- Убираем символ "#" если есть
    hex = hex:gsub("#", "")
    -- Проверяем длину
    if #hex ~= 6 then
        error("Недопустимый HEX цвет: должен содержать 6 символов")
    end
    -- Получаем компоненты R, G, B в десятичном формате
    local r = tonumber(hex:sub(1, 2), 16)
    local g = tonumber(hex:sub(3, 4), 16)
    local b = tonumber(hex:sub(5, 6), 16)
    -- Переводим в значения от 0 до 1 и возвращаем цвет
    return CS.UnityEngine.Color(r / 255, g / 255, b / 255, 1)
end

function Update()
if not isInitialized then
        return  -- Не выполняем логику, пока мод не инициализирован
    end

    UpdateSelectionOffset()
    local deltaTime = CS.UnityEngine.Time.deltaTime
    undoTimer = undoTimer + deltaTime

    if awaitingKeyBind then
    CheckMouseButtons()
    for keyCode, keyName in pairs(KeyCodeNames) do
        if keyCode ~= CS.UnityEngine.KeyCode.Mouse0
           and keyCode ~= CS.UnityEngine.KeyCode.Mouse1
           and keyCode ~= CS.UnityEngine.KeyCode.Mouse2 then
            if CS.UnityEngine.Input.GetKeyDown(keyCode) then
                AssignKeyToBind(keyCode)
                return
            end
        end
    end

    -- Обработка кнопок мыши для биндов
    if CS.UnityEngine.Input.GetMouseButtonDown(0) then
        AssignKeyToBind(CS.UnityEngine.KeyCode.Mouse0)
        return
    elseif CS.UnityEngine.Input.GetMouseButtonDown(1) then
        AssignKeyToBind(CS.UnityEngine.KeyCode.Mouse1)
        return
    elseif CS.UnityEngine.Input.GetMouseButtonDown(2) then
        AssignKeyToBind(CS.UnityEngine.KeyCode.Mouse2)
        return
    end
end

    if underBlockKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(underBlockKey) then
        SetBlockUnderPlayer()
    end

    --if CS.UnityEngine.Input.GetKeyDown(CS.UnityEngine.KeyCode.N) then
       -- ShowBlockFormUnderCrosshair()
    --end
    -- Отдельный блок для сообщения (один раз при нажатии)
if placeBlocksMode and CS.UnityEngine.Input.GetKeyDown(placeBlocksKey) then
    if AreAllSlidersAboveFive() then
        Chat:AddMessage("Невозможно установить блоки: значения сразу нескольких слайдеров 5 или больше", "[f54949]")
    end
end

hasShownNoBlockMessage = false  -- сбрасываем флаг каждый кадр

 -- Обработка установки блоков
if placeBlocksMode then
    if buildDestroyMode == "hold" and CS.UnityEngine.Input.GetKey(placeBlocksKey) then
        local currentTime = CS.UnityEngine.Time.time

        if currentTime - lastBlockActionTime >= CALL_INTERVAL then
            if not AreAllSlidersAboveFive() then
                SetBlock()
                lastBlockActionTime = currentTime
            end
        end
    elseif buildDestroyMode == "click" and CS.UnityEngine.Input.GetKeyDown(placeBlocksKey) then
        if AreAllSlidersAboveFive() then
            Chat:AddMessage("Невозможно установить блоки: значения сразу нескольких слайдеров 5 или больше", "[f54949]")
        else
            SetBlock()  -- Вызываем один раз при клике
        end
    end
end

-- Обработка удаления блоков
if destBlocksMode then
    if buildDestroyMode == "hold" and CS.UnityEngine.Input.GetKey(destBlocksKey) then
        local currentTime = CS.UnityEngine.Time.time

        if currentTime - lastDestroyActionTime >= CALL_INTERVAL then
            if not AreAllSlidersAboveFive() then
                DestroyBlock()  -- Ограниченный вызов при удержании
                lastDestroyActionTime = currentTime
            end
        end
    elseif buildDestroyMode == "click" and CS.UnityEngine.Input.GetKeyDown(destBlocksKey) then
        if AreAllSlidersAboveFive() then
            Chat:AddMessage("Невозможно удалить блоки: значения сразу нескольких слайдеров 5 или больше", "[f54949]")
        else
            DestroyBlock()  -- Вызываем один раз при клике
        end
    end
end

if destBlocksMode and CS.UnityEngine.Input.GetKeyDown(destBlocksKey) then
    if AreAllSlidersAboveFive() then
        Chat:AddMessage("Невозможно удалить блоки: значения сразу нескольких слайдеров 5 или больше", "[f54949]")
    end
end

    if CS.UnityEngine.Input.GetKeyDown(CS.UnityEngine.KeyCode.Insert) then
        showMenu = not showMenu
    end

    if selectAreaMode then
        -- Обработка нажатий клавиш для установки точек выделения
        HandleSelectionKeyInput()

        -- Удаление блоков в выделенной области по бинду 
        if deleteAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(deleteAreaKey) then
            StartDestroy()
        end

    --  копирование области по бинду 
        if copyAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(copyAreaKey) then
            CopySelection()
        end

        --  установка области по бинду 
        if pasteAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(pasteAreaKey) then
            StartPasting()
        end

        --  замена области по бинду 
        if replaceAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(replaceAreaKey) then
            StartReplacing()
        end

        --  заполнение области по бинду 
        if fillAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(fillAreaKey) then
            StartFilling()
        end

         -- отмена действия по бинду сукаываываоыв заеьался писаит
        if undoActionKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(undoActionKey) then
            StartUndo()
        end

         -- остановка процесса по бинду
        if stopProcessKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(stopProcessKey) then
            StopAllProcesses()
        end

          -- очистка области по бинду
        if clearAreaKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(clearAreaKey) then
            ClearSelectionAndReset()
            ClearAllDataTables()
        end

         -- поворот влево по бинду
        if rotateLeftKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(rotateLeftKey) then
            RotateCopiedBlocksLeft() 
        end

        -- поворот вправо по бинду
        if rotateRightKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(rotateRightKey) then
            RotateCopiedBlocksRight()
        end

         -- вкл/выкл отрисовки по бинду
        if drawKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(drawKey) then
            ToggleVisibility()
        end
    end

    if areaModedKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(areaModedKey) then
        ToggleStaticSelectionMode()
    end

    if undoTimer >= undoInterval then
        undoTimer = 0

        if isDeleting and blocksToDelete and #blocksToDelete > 0 then
        local endIndex = math.min(currentDeleteIndex + deleteSpeed - 1, #blocksToDelete)
        for i = currentDeleteIndex, endIndex do
        World:RemoveBlock(blocksToDelete[i].pos)
        blocksToDelete[i].processed = true  -- помечаем как обработанный
    end
    currentDeleteIndex = endIndex + 1
    if currentDeleteIndex > #blocksToDelete then
        isDeleting = false
    end
end

        if isReplacing and blocksToReplace and #blocksToReplace > 0 then
        local endIndex = math.min(currentReplaceIndex + deleteSpeed - 1, #blocksToReplace)
        for i = currentReplaceIndex, endIndex do
        local item = blocksToReplace[i]
        World:AddBlock(item.pos, item.blockType, item.blockForm)
        blocksToReplace[i].processed = true  -- помечаем как обработанный
    end
    currentReplaceIndex = endIndex + 1

    if currentReplaceIndex > #blocksToReplace then
        isReplacing = false
    end
end

    if isPlacing and blocksToPlace and #blocksToPlace > 0 then
    local endIndex = math.min(currentPlaceIndex + deleteSpeed - 1, #blocksToPlace)
    for i = currentPlaceIndex, endIndex do
        local item = blocksToPlace[i]
        World:AddBlock(item.pos, item.blockType, item.blockForm)
        blocksToPlace[i].processed = true  -- помечаем как обработанный
    end
    currentPlaceIndex = endIndex + 1
    if currentPlaceIndex > #blocksToPlace then
        isPlacing = false
        isPasteInProgress = false  -- сброс флага: вставка завершена
    end
end

  -- Постепенная отмена через чанки с учётом deleteSpeed
if isUndoing and undoChunks and currentChunkIndex <= #undoChunks then
    local currentChunk = undoChunks[currentChunkIndex]
    local processedInThisFrame = 0  -- счётчик обработанных блоков в текущем кадре

    -- Обрабатываем до deleteSpeed блоков за кадр
    while currentBlockIndexInChunk <= #currentChunk and processedInThisFrame < deleteSpeed do
        local block = currentChunk[currentBlockIndexInChunk]

        if IsPositionInBounds(block.pos) then
            local currentType = World:GetBlockType(block.pos)
            local currentForm = World:GetBlockKind(block.pos) or 0

            -- ПРОПУСКАЕМ, если блок уже соответствует сохранённому состоянию
            if currentType == block.blockType and currentForm == block.blockForm then
                currentBlockIndexInChunk = currentBlockIndexInChunk + 1
                processedInThisFrame = processedInThisFrame + 1
                goto continue_undo
            end

            -- Восстанавливаем только если состояние отличается
            if currentType ~= block.blockType or currentForm ~= block.form then
                if block.blockType == 0 then
                    World:RemoveBlock(block.pos)
                else
                    World:AddBlock(block.pos, block.blockType, block.blockForm)
                end
            end
        end

        currentBlockIndexInChunk = currentBlockIndexInChunk + 1
        processedInThisFrame = processedInThisFrame + 1

        ::continue_undo::
    end

    -- Если в текущем чанке закончились блоки, переходим к следующему
    if currentBlockIndexInChunk > #currentChunk then
        currentChunkIndex = currentChunkIndex + 1
        currentBlockIndexInChunk = 1
    end

    -- Завершение отмены
    if currentChunkIndex > #undoChunks then
        isUndoing = false
        undoChunks = nil
        lastActionBlocks = nil
        --Chat:AddMessage("Отмена последнего действия завершена", "[a455f2]")
    else
        -- Опциональное отображение прогресса (каждые 100 блоков или по желанию)
        local totalProcessed = (currentChunkIndex - 1) * #currentChunk + (currentBlockIndexInChunk - 1)
        if totalProcessed % 100 == 0 and totalProcessed > 0 then  -- показываем каждые 100 обработанных блоков
            LogChatMessage(string.format("Отмена: обработано %d из %d блоков", totalProcessed, #lastActionBlocks), "[888888]")
        end
    end
else
    -- Если по какой‑то причине обработка прервалась
    if isUndoing then
        isUndoing = false
        Chat:AddMessage("Ошибка при отмене действия", "[f54949]")
    end
end

        if isFilling and blocksToReplace and #blocksToReplace > 0 then
        local endIndex = math.min(currentFillIndex + deleteSpeed - 1, #blocksToReplace)
        for i = currentFillIndex, endIndex do
        local item = blocksToReplace[i]
        World:AddBlock(item.pos, item.blockType, item.blockForm)
        blocksToReplace[i].processed = true  -- помечаем как обработанный
    end
    currentFillIndex = endIndex + 1
    if currentFillIndex > #blocksToReplace then
        isFilling = false
    end
   end
end

if not selectAreaMode then
    isDeleting = false
    isReplacing = false
    isPlacing = false
    isFilling = false
end

-- Постепенное копирование области
if isCopying and copyChunks and currentCopyChunkIndex <= #copyChunks then
    local currentChunk = copyChunks[currentCopyChunkIndex]
    local processedInThisFrame = 0


    while currentBlockInCopyChunk <= #currentChunk and processedInThisFrame < copySpeed do
        local pos = currentChunk[currentBlockInCopyChunk]
        local blockType = World:GetBlockType(pos)

        if blockType ~= 0 then
            local blockForm = World.GetBlockKind and World:GetBlockKind(pos) or 0
            local offset = CS.UnityEngine.Vector3(
                pos.x - FirstPos.x,
                pos.y - FirstPos.y,
                pos.z - FirstPos.z
            )
            table.insert(buildData.blocks, {
                offset = offset,
                blockType = blockType,
                blockForm = blockForm
            })
        end

        currentBlockInCopyChunk = currentBlockInCopyChunk + 1
        processedInThisFrame = processedInThisFrame + 1
    end

    -- Переходим к следующему чанку, если текущий закончен
    if currentBlockInCopyChunk > #currentChunk then
        currentCopyChunkIndex = currentCopyChunkIndex + 1
        currentBlockInCopyChunk = 1
    end

    -- Завершение копирования
if currentCopyChunkIndex > #copyChunks then
    isCopying = false  -- критически важный сброс флага!
    copyChunks = nil
    allPositions = nil  -- освобождаем память
    totalCopyPositions = 0  -- сбрасываем счётчик
    isLargeCopy = false  -- сбрасываем флаг большой области
    useCopiedBlocks = true
    currentRotation = 0
    Chat:AddMessage(
        "Копирование завершено: " .. tostring(#buildData.blocks) .. " блоков.",
        "[a455f2]"
    )
else
    -- Расчёт процента завершения (только для больших областей)
    if isLargeCopy then
        local totalProcessed = (currentCopyChunkIndex - 1) * 100 + (currentBlockInCopyChunk - 1)
        local percentComplete = math.floor((totalProcessed / totalCopyPositions) * 100)  -- целое число процентов

        -- Получаем текущее время (в секундах с начала игры)
        local currentTime = CS.UnityEngine.Time.time

        -- Выводим прогресс каждые 2 секунды
        if currentTime - copyProgressLastUpdate >= copyProgressUpdateInterval then
            Chat:AddMessage(
                string.format("Копирование: %d%% завершено",
                    percentComplete, totalCopyPositions),
            "[b888cf]"
        )
            copyProgressLastUpdate = currentTime  -- обновляем время последнего сообщения
        end
    end
end
   else
    -- Обработка прерывания
    if isCopying then
        isCopying = false  -- сброс флага при прерывании
        allPositions = nil
        totalCopyPositions = 0
        isLargeCopy = false
        copyProgressLastUpdate = 0  -- сбрасываем таймер прогресса
        Chat:AddMessage("Ошибка при копировании", "[f54949]")
    end
end

     -- Если noclip активен, но у игрока больше нет прав, отключаем noclip
    if noclipActive and not (Player.IsAdmin or Player.IsModerator) then
        noclipActive = false
        local playerLayer = 2
        local terrainLayer = CS.UnityEngine.LayerMask.NameToLayer("Terrain")
        local entityLayer = CS.UnityEngine.LayerMask.NameToLayer("Entity")
        CS.UnityEngine.Physics.IgnoreLayerCollision(playerLayer, terrainLayer, false)
        CS.UnityEngine.Physics.IgnoreLayerCollision(playerLayer, entityLayer, false)
        print("Noclip disabled due to loss of permissions")
    end

    -- Обработка нажатия LeftControl для переключения noclip (уже с проверкой прав)
    if noclipToggle and noclipKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(noclipKey) then
        if not (Player.IsAdmin or Player.IsModerator) then
            return -- Нет прав включать noclip
        end

        noclipActive = not noclipActive
        local playerLayer = 2
        local terrainLayer = CS.UnityEngine.LayerMask.NameToLayer("Terrain")
        local entityLayer = CS.UnityEngine.LayerMask.NameToLayer("Entity")

        CS.UnityEngine.Physics.IgnoreLayerCollision(playerLayer, terrainLayer, noclipActive)
        CS.UnityEngine.Physics.IgnoreLayerCollision(playerLayer, entityLayer, noclipActive)

        if noclipActive then
            print("Noclip enabled")
        else
            print("Noclip disabled")
        end
    end

  local playerInput = FindPlayerInput()
if playerInput then
    if not Player.IsAdmin and not Player.IsModerator then
        return  -- Выходим, если нет прав
    end
    -- управление скоростью, как есть
    if speedToggle then
        playerInput:SetSpeedModifier(speedModifier)
    else
        playerInput:SetSpeedModifier(1)
    end

    if flyToggle ~= wasFlyToggle then
        playerInput:SendMessage("SetLadderBody", flyToggle)
        wasFlyToggle = flyToggle
    end
    -- если flyToggle выключен — не мешаем игре переключать полёт клавишей
end

    -- Проверка нажатия клавиши телепорта (если бинд назначен)
if teleportToggle and teleportKey ~= CS.UnityEngine.KeyCode.None and CS.UnityEngine.Input.GetKeyDown(teleportKey) then
    Teleport()
end
 -- Получаем время кадра правильно
    local deltaTime = CS.UnityEngine.Time.deltaTime  -- стандартный способ

    gcCooldown = gcCooldown - deltaTime

    -- Запуск раз в 10 секунд: начинаем новый цикл очистки
    if gcCooldown <= 0 and CS.UnityEngine.Time.time - lastGCRun >= GC_CHECK_INTERVAL then
        gcSteps = GC_TOTAL_STEPS  -- 5 шагов по 20 % мусора
        gcCooldown = GC_STEP_DELAY  -- шаг раз в 0.5 с
        lastGCRun = CS.UnityEngine.Time.time
    end

    --[[ Выполняем один шаг сбора мусора, если нужно
    if gcSteps > 0 then
        collectgarbage("step", 10)  -- небольшой шаг сбора мусора

        gcSteps = gcSteps - 1
        if gcSteps == 0 then
            gcCooldown = -1  -- сброс таймера, чтобы не запускать шаги до следующего цикла
        else
            gcCooldown = GC_STEP_DELAY  -- сбрасываем таймер для следующего шага
        end
    end]]
 CheckAndReportMemory()
end
